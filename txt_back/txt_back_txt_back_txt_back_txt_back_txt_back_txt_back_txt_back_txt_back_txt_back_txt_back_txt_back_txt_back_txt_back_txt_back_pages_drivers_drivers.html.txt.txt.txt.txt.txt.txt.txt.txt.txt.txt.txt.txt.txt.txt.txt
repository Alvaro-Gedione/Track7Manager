        function filterByStatus(status, element) {
            if (element) {

                element.classList.add('active');

            currentFilters.status = status;
            applyFilters();

            currentFilters.category = category;

        function filterByVehicle(vehicle) {
            currentFilters.vehicle = vehicle;

        function searchDrivers(query) {
            currentFilters.search = query.toLowerCase();
            applyFilters();

        function clearFilters() {
            currentFilters = {
                status: 'all',
                category: 'all',
                vehicle: 'all',
                search: '',
                hireMonth: ''
            };

            // Reset UI
            document.querySelectorAll('.filter-chip').forEach(el => el.classList.remove('active'));
            document.querySelector('.filter-chip[onclick*="all"]').classList.add('active');
            document.getElementById('categoryFilter').value = 'all';
            document.getElementById('vehicleFilter').value = 'all';
            document.getElementById('searchInput').value = '';
            document.getElementById('hireMonth').value = '';

            applyFilters();
        }

        function applyFilters() {
            filteredDrivers = drivers.filter(driver => {
                // Filtro de status
                if (currentFilters.status !== 'all' && driver.status !== currentFilters.status) {
                    return false;
                }

                // Filtro de categoria
                if (currentFilters.category !== 'all' && driver.category !== currentFilters.category) {
                    return false;
                }

                // Filtro de veículo
                if (currentFilters.vehicle === 'with' && !driver.vehicleId) return false;
                if (currentFilters.vehicle === 'without' && driver.vehicleId) return false;

                // Filtro de busca
                if (currentFilters.search) {
                    const searchStr = currentFilters.search.toLowerCase();
                    const matches =
                        driver.name.toLowerCase().includes(searchStr) ||
                        driver.cpf.includes(searchStr) ||
                        driver.cnh.includes(searchStr) ||
                        (driver.email && driver.email.toLowerCase().includes(searchStr));

                    if (!matches) return false;
                }

                // Filtro de mês de admissão
                if (currentFilters.hireMonth) {
                    const hireDate = driver.hireDate.substring(0, 7); // YYYY-MM
                    if (hireDate !== currentFilters.hireMonth) return false;
                }

                return true;
            });

            currentPage = 1;
            renderDrivers();
            updateStats();
        }

        }

        function getScoreColor(score) {
            if (score >= 90) return '#27ae60';

        function renderPagination() {
            const totalPages = Math.ceil(filteredDrivers.length / itemsPerPage);
            const pagination = document.getElementById('pagination');

            if (totalPages <= 1) {
                pagination.innerHTML = '';
                return;
            }

            let html = '<div class="pagination-controls">';

            // Botão anterior
            html += `
                <button class="page-btn" ${currentPage === 1 ? 'disabled' : ''} 
                        onclick="changePage(${currentPage - 1})">
                    <i class="fas fa-chevron-left"></i>
                </button>
            `;

            // Números das páginas
            for (let i = 1; i <= totalPages; i++) {
                if (
                    i === 1 ||
                    i === totalPages ||
                    (i >= currentPage - 2 && i <= currentPage + 2)
                ) {
                    html += `
                        <button class="page-btn ${i === currentPage ? 'active' : ''}" 
                                onclick="changePage(${i})">
                            ${i}
                        </button>
                    `;
                } else if (i === currentPage - 3 || i === currentPage + 3) {
                    html += '<span class="page-dots">...</span>';
                }
            }

            // Botão próximo
            html += `
                <button class="page-btn" ${currentPage === totalPages ? 'disabled' : ''} 
                        onclick="changePage(${currentPage + 1})">
                    <i class="fas fa-chevron-right"></i>
                </button>
            `;

            html += '</div>';
            pagination.innerHTML = html;
        }

            const fill = document.getElementById('progressFill');

            bar.classList.add('show');
            fill.style.width = value + '%';
        }

        function hideProgress() {
            const bar = document.getElementById('progressBar');
            bar.classList.remove('show');
        }

        function showToast(msg, duration = 3000) {
            const toast = document.getElementById('toast');
            toast.innerHTML = msg;
            toast.style.display = 'block';

            clearTimeout(window._toastTimer);
            window._toastTimer = setTimeout(() => {
                toast.style.display = 'none';
            }, duration);
        }
