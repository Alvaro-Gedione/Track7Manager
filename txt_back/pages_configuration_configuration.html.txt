        function switchTab(el, tabName) {
            // Atualiza as classes das tabs
            document.querySelectorAll('.breadcrumb-bar .tab').forEach(t => t.classList.remove('active'));
            el.classList.add('active');

            currentTab = tabName;

            // Atualiza visibilidade dos conteúdos principais
            document.querySelectorAll('.tab-content').forEach(c => {
                c.classList.toggle('active', c.id.includes(tabName));
            });

            // Controla visibilidade das listas laterais
            const clientsList = document.getElementById('clientsList');
            const groupsList = document.getElementById('groupsList');

            if (clientsList) clientsList.classList.toggle('active', tabName === 'clients');
            if (groupsList) groupsList.classList.toggle('active', tabName === 'groups');

            // Atualiza placeholder da busca
            const searchInput = document.getElementById('searchInput');
            if (searchInput) {
                searchInput.placeholder = tabName === 'groups' ? 'Buscar grupos...' : 'Buscar clientes...';
                searchInput.value = '';
            }

            // Garante que o conteúdo correto é exibido
            if (tabName === 'clients') {
                // Se está na aba de clientes, garante que algum cliente está selecionado
                if (currentClientId && clients.find(c => c.id === currentClientId)) {
                    selectClient(currentClientId);
                } else if (clients.length > 0) {

            }

        }

                    selectGroup(groups[0].id);
                } else {
                    const tbody = document.getElementById('equipmentTableBody');
                    if (tbody) tbody.innerHTML = '<tr><td colspan="6" style="text-align: center;">Nenhum grupo encontrado</td></tr>';

                    if (cards) cards.innerHTML = '<div style="text-align: center; padding: 20px;">Selecione um grupo para ver as configurações.</div>';
                }
            } else {
                // Se uma empresa específica está selecionada

                if (selectedClient) {
                    const firstGroupForClient = groups.find(g => g.companyId === selectedClient.id);

        function filterClientsList(companyId) {
            const clientItems = document.querySelectorAll('.client-item');

            if (companyId) {
                clientItems.forEach(item => {
                    // Extrai o ID do cliente do elemento (você pode adicionar um atributo data-id)
                    // Vamos modificar o loadClients para adicionar um atributo data-id
                    const clientId = parseInt(item.getAttribute('data-client-id'));
                    if (clientId === companyId) {
                        item.style.display = 'block';
                    } else {
                        item.style.display = 'none';
                    }
                });
            } else {
                // Mostra todos
                clientItems.forEach(item => {
                    item.style.display = 'block';
                });
            }
        }

        function filterList(query) {
            const items = currentTab === 'groups' ? document.querySelectorAll('.group-item') : document.querySelectorAll('.client-item');

            items.forEach(item => {
                const text = item.textContent.toLowerCase();
                item.style.display = text.includes(query.toLowerCase()) ? 'block' : 'none';
            });
        }

            document.getElementById('configReports').checked = true;
            document.getElementById('configAlerts').checked = true;
            document.getElementById('groupModalTitle').textContent = 'Novo Grupo de Configuração';
        }

        function saveGroup() {
            const name = document.getElementById('groupName').value;
            if (!name) {
                showToast('Nome do grupo é obrigatório', 'error');
                return;
            }

            const companyId = parseInt(document.getElementById('groupCompany').value);
            const desc = document.getElementById('groupDesc').value;

            // Valida se um cliente foi selecionado
            if (!companyId) {
                showToast('Selecione uma empresa para o grupo', 'error');
                return;
            }

            groups.push({
                id: groups.length + 1,
                name: name,
                description: desc,
                companyId: companyId, // Salva o ID, não o nome
                configs: {
                    tracking: document.getElementById('configTracking').checked,
                    reports: document.getElementById('configReports').checked,
                    alerts: document.getElementById('configAlerts').checked
                },
                equipmentCount: 0
            });

            loadGroups();
            closeModal('groupModal');
            showToast('Grupo criado com sucesso!');
        }

        function closeModal(id) {
            document.getElementById(id).classList.remove('open');
        }

        function showToast(message, type = 'success') {
            const toast = document.getElementById('toast');
            const icon = toast.querySelector('i');

            if (type === 'success') {
                icon.className = 'fas fa-check-circle';
                toast.style.background = 'var(--brand-orange)';
            } else {
                icon.className = 'fas fa-exclamation-circle';
                toast.style.background = '#e74c3c';
            }

            document.getElementById('toastMessage').textContent = message;
            toast.style.display = 'flex';

            setTimeout(() => {
                toast.style.display = 'none';
            }, 3000);
        }

        function toggleGroupMenu(id) {
            document.querySelectorAll('[id^="groupMenu-"]').forEach(menu => {
                if (menu.id !== `groupMenu-${id}`) menu.style.display = 'none';
            });
            const menu = document.getElementById(`groupMenu-${id}`);
            menu.style.display = menu.style.display === 'none' ? 'block' : 'none';
        }

        function toggleClientMenu(id) {
            document.querySelectorAll('[id^="clientMenu-"]').forEach(menu => {
                if (menu.id !== `clientMenu-${id}`) menu.style.display = 'none';
            });
            const menu = document.getElementById(`clientMenu-${id}`);
            menu.style.display = menu.style.display === 'none' ? 'block' : 'none';
        }

        function editGroup(id) {
            const group = groups.find(g => g.id === id);
            if (group) {
                document.getElementById('groupName').value = group.name;
                document.getElementById('groupDesc').value = group.description || '';
                document.getElementById('groupCompany').value = group.companyId; // Carrega pelo ID
                document.getElementById('configTracking').checked = group.configs?.tracking || false;
                document.getElementById('configReports').checked = group.configs?.reports || false;
                document.getElementById('configAlerts').checked = group.configs?.alerts || false;
                document.getElementById('groupModalTitle').textContent = 'Editar Grupo';
                document.getElementById('groupModal').classList.add('open');
            }
            toggleGroupMenu(id);
        }

        function deleteGroup(id) {
            if (confirm('Tem certeza que deseja excluir este grupo?')) {
                groups = groups.filter(g => g.id !== id);
                loadGroups();
                showToast('Grupo excluído com sucesso!');
            }
            toggleGroupMenu(id);
        }

        function duplicateGroup(id) {
            const group = groups.find(g => g.id === id);
            if (group) {
                const newGroup = { ...group, id: groups.length + 1, name: `${group.name} (cópia)` };
                groups.push(newGroup);
                loadGroups();
                showToast('Grupo duplicado com sucesso!');
            }
            toggleGroupMenu(id);
        }

        function editClient(id) {
            const client = clients.find(c => c.id === id);
            if (client) showToast(`Editando cliente: ${client.tradeName}`);
            toggleClientMenu(id);
        }

        function deleteClient(id) {
            if (confirm('Tem certeza que deseja excluir este cliente?')) {
                clients = clients.filter(c => c.id !== id);
                loadClients();
                showToast('Cliente excluído com sucesso!');
            }
            toggleClientMenu(id);
        }

        function viewClientDetails(id) {
            const client = clients.find(c => c.id === id);
            if (client) showToast(`Detalhes do cliente: ${client.tradeName}`);
            toggleClientMenu(id);
        }

        // ADICIONE ESTAS FUNÇÕES ANTES DO DOMContentLoaded

        // Função para salvar a organização selecionada no localStorage
        function saveSelectedOrganization(companyId) {
            if (companyId) {
                localStorage.setItem('selectedOrganization', companyId);
                console.log('Organização salva:', companyId);
            } else {
                localStorage.removeItem('selectedOrganization');
                console.log('Organização removida (todas empresas)');
            }
        }

        // Função para carregar a organização salva
        function loadSelectedOrganization() {
            const savedCompanyId = localStorage.getItem('selectedOrganization');
            const companySelect = document.getElementById('companySelect');

            if (savedCompanyId && companySelect) {
                // Verifica se o valor salvo existe no select
                const optionExists = Array.from(companySelect.options).some(opt => opt.value === savedCompanyId);

                if (optionExists) {
                    companySelect.value = savedCompanyId;
                    // Aplica o filtro com a organização salva
                    if (typeof filterByCompany === "function") {
                        filterByCompany(savedCompanyId);
                    }
                    console.log('Organização carregada:', savedCompanyId);
                    return savedCompanyId;
                }
            }

            // Se não houver organização salva ou ela não existir mais, mantém "Todas as Empresas"
            if (companySelect) {
                companySelect.value = "";
                if (typeof filterByCompany === "function") {
                    filterByCompany("");
                }
            }
            return null;
        }

        // MODIFIQUE A FUNÇÃO filterByCompany para salvar a seleção
        function filterByCompany(id) {
            if (id) {
                const companyId = parseInt(id);

                if (currentTab === 'clients') {
                    // Agora loadClients já faz o trabalho de "arrancar" os outros da lista
                    loadClients();
                    selectClient(companyId);
                } else if (currentTab === 'groups') {
                    currentClientId = companyId;
                    updateGroupsForSelectedClient();
                    updateClientMetricsInGroupsTab();
                }

                // SALVA A ORGANIZAÇÃO NO LOCALSTORAGE
                saveSelectedOrganization(companyId);

                showToast(`Filtrando por: ${clients.find(c => c.id == id)?.tradeName}`);
            } else {
                // Quando "Todas as Empresas" é selecionado
                showToast("Visualizando todas as organizações");

                // SALVA "VAZIO" NO LOCALSTORAGE (remove a organização salva)
                saveSelectedOrganization(null);

                loadClients(); // Recarrega todos
                loadGroups();  // Recarrega todos

                if (clients.length > 0) selectClient(clients[0].id);
            }
        }

        // MODIFIQUE O DOMContentLoaded para carregar a organização salva
        document.addEventListener('DOMContentLoaded', () => {
            const savedTheme = localStorage.getItem('theme') || 'dark';
            document.body.className = `theme-${savedTheme}`;

            loadCompanies();
            loadGroups();
            loadClients();
            loadTags();
            setupEventListeners();

            // CARREGA A ORGANIZAÇÃO SALVA PRIMEIRO
            loadSelectedOrganization();

            // NÃO força a seleção da aba de clientes mais
            // Mantém a aba que estiver ativa no HTML (que é 'clients' por padrão)

            // Seleciona o primeiro cliente por padrão APENAS SE NÃO HOUVER ORGANIZAÇÃO SALVA
            const savedOrg = localStorage.getItem('selectedOrganization');
            if (!savedOrg && clients.length > 0) {
                selectClient(clients[0].id);
            }

            // Se a aba de grupos estiver ativa, atualiza as métricas
            if (currentTab === 'groups') {
                updateClientMetricsInGroupsTab();
            }
        });

        // MODIFIQUE O EVENT LISTENER DO organizationChanged para usar localStorage
        document.addEventListener('organizationChanged', (e) => {
            const companyId = e.detail.companyId;
            const companySelect = document.getElementById('companySelect');

            if (companySelect && companySelect.value !== companyId) {
                companySelect.value = companyId;

                // SALVA NO LOCALSTORAGE
                saveSelectedOrganization(companyId);

                if (typeof filterByCompany === "function") {
                    filterByCompany(companyId);
                }
            }
        });
