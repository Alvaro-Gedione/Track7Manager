        function renderDrivers() {
            const container = document.getElementById('driversContainer');

            if (filteredDrivers.length === 0) {
                container.innerHTML = `
                    <div class="empty-state">
                        <i class="fas fa-user-slash"></i>
                        <h3>Nenhum motorista encontrado</h3>
                        <p>Tente ajustar os filtros ou cadastrar um novo motorista.</p>
                        <button class="btn-primary" onclick="openDriverModal()">
                            <i class="fas fa-plus"></i> Novo Motorista
                        </button>
                    </div>
                `;
                document.getElementById('pagination').innerHTML = '';
                return;
            }

            const start = (currentPage - 1) * itemsPerPage;
            const end = Math.min(start + itemsPerPage, filteredDrivers.length);
            const pageDrivers = filteredDrivers.slice(start, end);

            if (currentViewMode === 'grid') {
                renderGridView(container, pageDrivers);
            } else {
                renderTableView(container, pageDrivers);
            }

            renderPagination();
        }

        function getScoreColor(score) {
            if (score >= 90) return '#27ae60';
            if (score >= 75) return '#f39c12';
            return '#e74c3c';
        }

        function renderGridView(container, driversList) {
            container.className = 'drivers-grid';
            let html = '';

            driversList.forEach(driver => {
                const vehicle = vehicles.find(v => v.id === driver.vehicleId);
                const statusClass = { 'active': 'status-active', 'inactive': 'status-inactive', 'on_break': 'status-break' }[driver.status] || '';
                const statusText = { 'active': 'Ativo', 'inactive': 'Inativo', 'on_break': 'Em Pausa' }[driver.status] || '';

        function renderTableView(container, driversList) {

        function renderPagination() {
            const totalPages = Math.ceil(filteredDrivers.length / itemsPerPage);
            const pagination = document.getElementById('pagination');

            if (totalPages <= 1) {
                pagination.innerHTML = '';
                return;
            }

            let html = '<div class="pagination-controls">';
            html += `<button class="page-btn" ${currentPage === 1 ? 'disabled' : ''} onclick="changePage(${currentPage - 1})"><i class="fas fa-chevron-left"></i></button>`;

            for (let i = 1; i <= totalPages; i++) {
                if (i === 1 || i === totalPages || (i >= currentPage - 2 && i <= currentPage + 2)) {
                    html += `<button class="page-btn ${i === currentPage ? 'active' : ''}" onclick="changePage(${i})">${i}</button>`;
                } else if (i === currentPage - 3 || i === currentPage + 3) {
                    html += '<span class="page-dots">...</span>';
                }
            }

            html += `<button class="page-btn" ${currentPage === totalPages ? 'disabled' : ''} onclick="changePage(${currentPage + 1})"><i class="fas fa-chevron-right"></i></button>`;
            html += '</div>';
            pagination.innerHTML = html;
        }

        function changePage(page) {
            currentPage = page;
            renderDrivers();
        }

        function toggleViewMode() {
            currentViewMode = currentViewMode === 'grid' ? 'table' : 'grid';
            const icon = document.getElementById('viewModeIcon');
            icon.className = currentViewMode === 'grid' ? 'fas fa-th-large' : 'fas fa-list';
            renderDrivers();
        }

        // ==================== CRUD ====================
        function openDriverModal(driverId = null) {
            selectedDriverId = driverId;
            if (driverId) {
                const driver = drivers.find(d => d.id === driverId);
                if (driver) {
                    document.getElementById('modalTitle').innerHTML = '<i class="fas fa-user-edit"></i> <span>Editar Motorista</span>';
                    document.getElementById('driverName').value = driver.name;
                    document.getElementById('driverCpf').value = driver.cpf;
                    document.getElementById('driverCnh').value = driver.cnh;
                    document.getElementById('driverBirth').value = driver.birth || '';
                    document.getElementById('driverHireDate').value = driver.hireDate || '';
                    document.getElementById('driverCategory').value = driver.category;
                    document.getElementById('driverVehicle').value = driver.vehicleId || '';
                    document.getElementById('driverEmail').value = driver.email || '';
                    document.getElementById('driverPhone').value = driver.phone || '';
                    document.getElementById('driverAddress').value = driver.address || '';
                    document.getElementById('driverStatus').value = driver.status;
                    document.getElementById('driverCnhExpiry').value = driver.cnhExpiry || '';
                    document.getElementById('driverMedical').value = driver.medical || '';
                }
            } else {
                document.getElementById('modalTitle').innerHTML = '<i class="fas fa-user-plus"></i> <span>Novo Motorista</span>';
                document.getElementById('driverForm').reset();
            }
            document.getElementById('driverModal').classList.add('open');
        }

        function closeDriverModal() {
            document.getElementById('driverModal').classList.remove('open');
            selectedDriverId = null;
        }

            event.preventDefault();
            showProgress(50);

                setTimeout(() => {

                    showToast(selectedDriverId ? 'Motorista atualizado com sucesso!' : 'Motorista cadastrado com sucesso!');

        function deleteDriver(driverId) {
            if (confirm('Tem certeza que deseja excluir este motorista?')) {
                showProgress(50);
                setTimeout(() => {
                    drivers = drivers.filter(d => d.id !== driverId);
                    applyFilters();
                    showProgress(100);
                    setTimeout(() => { hideProgress(); showToast('Motorista removido com sucesso!'); }, 200);
                }, 500);
            }
        }

        function showProgress(value) {
            const bar = document.getElementById('progressBar');
            const fill = document.getElementById('progressFill');
            bar.classList.add('show');
            fill.style.width = value + '%';
        }

        function hideProgress() { document.getElementById('progressBar').classList.remove('show'); }

        function showToast(msg) {
            const toast = document.getElementById('toast');
            toast.textContent = msg;

            clearTimeout(window._toastTimer);
            window._toastTimer = setTimeout(() => { toast.style.display = 'none'; }, 3000);
