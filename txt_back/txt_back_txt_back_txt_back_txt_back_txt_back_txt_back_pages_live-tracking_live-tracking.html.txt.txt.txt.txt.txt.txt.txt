        async function fetchAllVehiclesPosition() {
            if (!isTracking) return;

            for (const vehicle of vehicles) {
                if (vehicle.status !== 'off') {
                    await fetchVehiclePosition(vehicle.id);
                    // Espera 200ms entre cada veículo para evitar o erro 1003 (api_concurrent)
                    await new Promise(resolve => setTimeout(resolve, 200));
                }
            }

            renderAllMarkers();
            renderVehicleList(document.getElementById('vehicleSearch')?.value || '', currentStatusFilter);
            updateLastUpdateTime();
        }

        async function fetchVehiclePosition(unitId) {
            try {
                // Removendo o cálculo de tempo para não filtrar. 
                // Queremos o último registro, não importa se foi hoje ou ontem.
                const response = await fetch(`${API}/messages`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        unitId: unitId,
                        timeFrom: 0,
                        timeTo: Math.floor(Date.now() / 1000),
                        msgType: 1,
                        maxCount: 1,
                        flags: 0x0001 // Se sua API suportar, isso indica "última mensagem"
                    })
                });

                const data = await response.json();

                if (data.messages && data.messages.length > 0) {
                    // Pegamos a primeira mensagem do array (que será a mais nova)
                    const latestMsg = data.messages[0];
                    updateVehicleFromMessage(unitId, latestMsg);
                }
            } catch (error) {
                console.error(`Erro na telemetria do veículo ${unitId}:`, error);
            }
        }

        function updateVehicleFromMessage(unitId, msg) {
            const vehicle = vehicles.find(v => v.id == unitId);
            if (!vehicle || !msg.pos) return;

            // 1. Atualiza apenas os dados necessários
            vehicle.lat = msg.pos.y;
            vehicle.lng = msg.pos.x;
            vehicle.speed = Math.round(msg.pos.s || 0);
            vehicle.lastUpdate = new Date(msg.t * 1000).toLocaleTimeString('pt-BR');

            // 2. MOVE O MARCADOR NO MAPA SEM RERENDERIZAR TUDO
            if (vehicleMarkers[unitId]) {
                // Leaflet move o marcador suavemente se houver transição CSS
                vehicleMarkers[unitId].setLatLng([vehicle.lat, vehicle.lng]);

                // Atualiza o conteúdo do popup sem fechá-lo
                if (vehicleMarkers[unitId].getPopup()) {
                    vehicleMarkers[unitId].setPopupContent(createPopupContent(vehicle));
                }
            } else {
                // Se o marcador não existir (primeira carga), cria ele
                renderAllMarkers();
            }
        }

        function startRealTimeTracking() {
            // Definimos o intervalo para 60 segundos (1 minuto) para evitar limites da API
            const INTERVALO_UM_MINUTO = 60000;

            if (trackingInterval) clearInterval(trackingInterval);

            // Primeira execução imediata ao carregar
            if (isTracking) {
                fetchAllVehiclesPosition();
            }

            trackingInterval = setInterval(() => {
                if (isTracking) {
                    console.log('🔄 Iniciando atualização de telemetria (Intervalo: 1 min)');
                    fetchAllVehiclesPosition();
                }
            }, INTERVALO_UM_MINUTO);

            // O relógio da interface continua atualizando a cada segundo normalmente
            setInterval(updateClock, 1000);
        }

            lastUpdateTime = Date.now();
            const date = new Date();
            document.getElementById('lastUpdate').innerHTML = `<i class="far fa-clock" style="margin-right: 4px;"></i> ${date.toLocaleTimeString()}`;
        }

        function updateClock() {
            const now = new Date();
            document.getElementById('lastUpdate').innerHTML = `<i class="far fa-clock" style="margin-right: 4px;"></i> ${now.toLocaleTimeString()}`;
        }

        function initMap() {
            // Centralizar em Santa Rita do Sapucaí como padrão
            map = L.map('realMap', { zoomControl: false }).setView([-22.2568, -45.7035], 13);

            // Configurar camada base baseada no tema atual
            updateMapTheme();

            // Se já houver veículos, centralizar neles
            if (vehicles.length > 0) {
                centerMapOnVehicles();
            }
        }

        function updateMapTheme() {
            // Remove camadas de tiles existentes para evitar sobreposição
            map.eachLayer((layer) => {
                if (layer instanceof L.TileLayer) {
                    map.removeLayer(layer);
                }
            });

            const currentTheme = document.body.classList.contains('theme-light') ? 'light' : 'dark';

            if (currentTheme === 'light') {
                L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', {
                    attribution: '© OpenStreetMap, © CartoDB'
                }).addTo(map);
            } else {
                L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
                    attribution: '© OpenStreetMap, © CartoDB'
                }).addTo(map);
            }
        }

        function renderAllMarkers() {
            // Remover marcadores existentes
            Object.values(vehicleMarkers).forEach(marker => map.removeLayer(marker));
            vehicleMarkers = {};

            // Filtrar veículos sem coordenadas
            const vehiclesWithPosition = vehicles.filter(v => v.lat != null && v.lng != null);

            vehiclesWithPosition.forEach(vehicle => {
                const marker = createVehicleMarker(vehicle);
                marker.addTo(map);
                vehicleMarkers[vehicle.id] = marker;
            });

            // Centralizar mapa se houver veículos
            if (vehiclesWithPosition.length > 0 && !selectedVehicleId) {
                centerMapOnVehicles();
            }
        }

        function centerMapOnVehicles() {
            const vehiclesWithPosition = vehicles.filter(v => v.lat != null && v.lng != null);

            if (vehiclesWithPosition.length === 0) return;

            if (vehiclesWithPosition.length === 1) {
                map.setView([vehiclesWithPosition[0].lat, vehiclesWithPosition[0].lng], 15);
            } else {
                const bounds = L.latLngBounds(vehiclesWithPosition.map(v => [v.lat, v.lng]));
                map.fitBounds(bounds, { padding: [50, 50] });
            }
        }

        function renderVehicleList(filter = '', statusFilter = 'all') {
            const list = document.getElementById('vehicleList');
            if (!list) return;

            list.innerHTML = '';

            const filtered = vehicles.filter(v => {
                const matchText = v.name.toLowerCase().includes(filter.toLowerCase()) ||
                    v.plate.toLowerCase().includes(filter.toLowerCase()) ||
                    (v.driver && v.driver.toLowerCase().includes(filter.toLowerCase()));
                const matchStatus = statusFilter === 'all' || v.status === statusFilter;
                return matchText && matchStatus;
            });

            if (filtered.length === 0) {
                list.innerHTML = '<div style="padding:30px;text-align:center;color:var(--brand-silver);"><i class="fas fa-search" style="font-size:24px;margin-bottom:10px;opacity:0.5;"></i><br>Nenhum veículo encontrado.</div>';
                return;
            }

            filtered.forEach((v, index) => {
                const div = document.createElement('div');
                div.className = 'vehicle-item' + (v.id === selectedVehicleId ? ' selected' : '');
                div.style.animationDelay = (index * 0.05) + 's';
                div.onclick = () => selectVehicle(v.id);

                const statusColorMap = {
                    active: 'var(--status-active)',
                    idle: 'var(--status-idle)',
                    off: 'var(--status-off)',
                    alert: 'var(--status-alert)'
                };

                div.innerHTML = `
            <div class="vi-top">
                <span class="vi-status-dot" style="background:${statusColorMap[v.status] || '#999'};"></span>
                <span class="vi-name">${v.name}</span>
                <span class="vi-plate">${v.plate}</span>
            </div>
            <div class="vi-mid">
                <span class="vi-driver"><i class="far fa-user" style="margin-right:3px;"></i> ${v.driver || '—'}</span>
                ${v.event ? `<span class="vi-event-badge"><i class="fas fa-exclamation-triangle" style="margin-right:2px;"></i> ${v.event}</span>` : ''}
            </div>
            <div class="vi-bottom">
                <span class="vi-location"><i class="fas fa-map-marker-alt" style="margin-right:3px;font-size:9px;"></i> 
                    ${v.lat ? `${v.lat.toFixed(6)}, ${v.lng?.toFixed(6)}` : v.location}
                </span>
                <span class="vi-speed ${v.speed > v.maxSpeed ? 'speeding' : ''}">
                    ${v.status === 'off' ? '—' : '<i class="fas fa-tachometer-alt" style="margin-right:2px;"></i> ' + v.speed + ' km/h'}
                </span>
            </div>
            <div style="font-size:9px;color:var(--brand-silver);text-align:right;margin-top:2px;">
                <i class="far fa-clock"></i> ${v.lastUpdate}
            </div>
        `;
                list.appendChild(div);
            });
        }

        function updateStats() {
            const active = vehicles.filter(v => v.status === 'active').length;
            const idle = vehicles.filter(v => v.status === 'idle').length;
            const off = vehicles.filter(v => v.status === 'off').length;
            const alert = vehicles.filter(v => v.event).length;

            document.getElementById('countAll').textContent = vehicles.length;
            document.getElementById('countActive').textContent = active;
            document.getElementById('countIdle').textContent = idle;
            document.getElementById('countOff').textContent = off;

            // Adicionar contador de alertas se existir
            const alertPill = document.querySelector('.stat-pill[onclick*="alert"]');
            if (alertPill) {
                const alertNum = alertPill.querySelector('.stat-num');
                if (alertNum) alertNum.textContent = alert;
            }
        }

        function selectVehicle(id) {
            selectedVehicleId = id;
            renderVehicleList(
                document.getElementById('vehicleSearch')?.value || '',
                currentStatusFilter
            );

            const vehicle = vehicles.find(v => v.id === id);
            if (!vehicle) return;

            // Centralizar mapa no veículo selecionado
            if (vehicle.lat && vehicle.lng) {
                map.setView([vehicle.lat, vehicle.lng], 16);

                // Abrir popup
                if (vehicleMarkers[vehicle.id]) {
                    vehicleMarkers[vehicle.id].openPopup();
                }
            }

            showPopup(vehicle);
        }

        function filterByStatus(status, el) {
            currentStatusFilter = status;
            document.querySelectorAll('.stat-pill').forEach(p => p.classList.remove('active-filter'));
            el.classList.add('active-filter');
            renderVehicleList(
                document.getElementById('vehicleSearch')?.value || '',
                status
            );

            // Animate stats
            const statNum = el.querySelector('.stat-num');
            statNum.classList.add('loading');
            setTimeout(() => statNum.classList.remove('loading'), 500);

            showToast(`Filtrando: ${el.textContent.trim()}`);
        }

        function filterVehicles(q) {
            renderVehicleList(q, currentStatusFilter);
        }

        function startTracking() {
            setInterval(() => {
                // Simulação de movimento aleatório (pequenos desvios lat/lng)
                const moveLat = (Math.random() - 0.5) * 0.001;
                const moveLng = (Math.random() - 0.5) * 0.001;

                vehicleData.lat += moveLat;
                vehicleData.lng += moveLng;
                vehicleData.speed = Math.floor(Math.random() * (80 - 40 + 1)) + 40;

                // Atualiza posição do marcador
                const newPos = [vehicleData.lat, vehicleData.lng];
                vehicleMarker.setLatLng(newPos);

                // Faz o mapa seguir o veículo suavemente
                map.panTo(newPos);

                // Atualiza a barra de status e lista lateral
                updateUI();
            }, 3000); // Atualiza a cada 3 segundos
        }

        function updateUI() {

            const now = new Date();
            document.getElementById('lastUpdate').innerHTML = `<i class="far fa-clock"></i> ${now.toLocaleTimeString()}`;

            // Atualiza Lista Lateral (exemplo simplificado)
            const list = document.getElementById('vehicleList');
            list.innerHTML = `
                <div class="vehicle-item selected">

                        <span class="vi-status-dot" style="background: var(--status-active);"></span>

                        <span class="vi-plate">${vehicleData.plate}</span>
                    </div>
                    <div class="vi-bottom">
                        <span class="vi-location">Lat: ${vehicleData.lat.toFixed(4)}...</span>
                        <span class="vi-speed"><i class="fas fa-tachometer-alt"></i> ${vehicleData.speed} km/h</span>
                    </div>
                </div>
            `;
        }

            } else {

        const statusColor = {
            active: 'var(--status-active)',
            idle: 'var(--status-idle)',
            off: 'var(--status-off)',
        };



        /* RENDER MAP MARKERS */
        function renderMarkers() {
            const container = document.getElementById('markersContainer');
            container.innerHTML = '';
            vehicles.forEach(v => {
                const div = document.createElement('div');
                div.className = 'vehicle-marker';
                div.id = 'marker-' + v.id;
                div.style.left = v.mapX + '%';
                div.style.top = v.mapY + '%';
                div.onclick = (e) => {
                    e.stopPropagation();
                    selectVehicle(v.id);
                };

                const borderColor = statusColor[v.status];
                div.innerHTML = `
                    <div class="marker-label">${v.plate}</div>
                    <div class="marker-icon" style="background:${borderColor};border-color:${borderColor};">
                        ${v.icon}
                    </div>
                `;
                container.appendChild(div);
            });
        }

        /* SHOW POPUP */
        function showPopup(v) {
            const popup = document.getElementById('vehiclePopup');
            document.getElementById('popupTitle').innerHTML = `<i class="fas fa-bus-alt" style="margin-right: 8px;"></i> ${v.name}`;

            const speedClass = v.speed > v.maxSpeed ? 'red' : 'green';
            const ignClass = v.ignition === 'Ligado' ? 'green' : '';

            document.getElementById('popupBody').innerHTML = `
                <div class="popup-row">
                    <span class="lbl"><i class="fas fa-id-card" style="margin-right:5px;"></i> Placa</span>
                    <span class="val">${v.plate}</span>
                </div>
                <div class="popup-row">
                    <span class="lbl"><i class="fas fa-user" style="margin-right:5px;"></i> Motorista</span>
                    <span class="val">${v.driver}</span>
                </div>
                <div class="popup-row">
                    <span class="lbl"><i class="fas fa-tachometer-alt" style="margin-right:5px;"></i> Velocidade</span>
                    <span class="val ${speedClass}">${v.status === 'off' ? '—' : v.speed + ' km/h'}</span>
                </div>
                <div class="popup-row">
                    <span class="lbl"><i class="fas fa-key" style="margin-right:5px;"></i> Ignição</span>
                    <span class="val ${ignClass}">${v.ignition}</span>
                </div>
                <div class="popup-row">
                    <span class="lbl"><i class="fas fa-map-marker-alt" style="margin-right:5px;"></i> Localização</span>
                    <span class="val" style="font-size:10.5px;">${v.location}</span>
                </div>
                <div class="popup-row">
                    <span class="lbl"><i class="fas fa-road" style="margin-right:5px;"></i> Hodômetro</span>
                    <span class="val">${v.odometer}</span>
                </div>
                <div class="popup-row">
                    <span class="lbl"><i class="far fa-clock" style="margin-right:5px;"></i> Atualização</span>
                    <span class="val">${v.lastUpdate}</span>
                </div>
                ${v.event ? `
                <div class="popup-row">
                    <span class="lbl"><i class="fas fa-exclamation-triangle" style="margin-right:5px;"></i> Evento</span>
                    <span class="val red">⚠ ${v.event}</span>
                </div>` : ''}
            `;

            // Position popup near the marker
            const marker = document.getElementById('marker-' + v.id);
            const mapArea = document.getElementById('mapArea');
            const mapRect = mapArea.getBoundingClientRect();
            const markerRect = marker.getBoundingClientRect();

            let left = markerRect.left - mapRect.left + 40;
            let top = markerRect.top - mapRect.top - 20;

            // Clamp so it doesn't go off map
            if (left + 280 > mapRect.width) left = markerRect.left - mapRect.left - 290;
            if (top + 320 > mapRect.height) top = mapRect.height - 330;
            if (top < 0) top = 10;

            popup.style.left = left + 'px';
            popup.style.top = top + 'px';
            popup.classList.add('show');
        }

        /* MAP TYPE */
        function setMapType(btn) {
            document.querySelectorAll('.map-type-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            showToast('Mapa alterado para: ' + btn.textContent.trim());

            mapBg.style.animation = 'none';

            if (type.includes('Satélite')) {
                mapBg.style.backgroundColor = '#1a3a2a';
            } else if (type.includes('Tráfego')) {
                mapBg.style.backgroundColor = '#2a1a1a';

                mapBg.style.backgroundColor = 'var(--map-bg)';
            }
        }

        function showToast(msg, duration = 3000) {
            const toast = document.getElementById('toast');
            if (!toast) return;

            toast.innerHTML = msg;
            toast.style.display = 'block';

            clearTimeout(window._toastTimer);
            window._toastTimer = setTimeout(() => {
                toast.style.display = 'none';
            }, duration);
        }

        /* ANIMATE ACTIVE MARKERS (simulate movement) */
        function startAnimations() {
            if (animationInterval) clearInterval(animationInterval);

            animationInterval = setInterval(() => {
                vehicles.forEach(v => {
                    if (v.status !== 'active') return;

                    // Smooth movement
                    const dx = (Math.random() - 0.5) * 0.8;
                    const dy = (Math.random() - 0.5) * 0.8;
                    v.mapX = Math.min(90, Math.max(5, v.mapX + dx));
                    v.mapY = Math.min(85, Math.max(10, v.mapY + dy));

                    const marker = document.getElementById('marker-' + v.id);
                    if (marker) {
                        marker.style.left = v.mapX + '%';
                        marker.style.top = v.mapY + '%';

                        // Add bounce effect
                        marker.style.transform = 'scale(1.1)';
                        setTimeout(() => {
                            if (marker) marker.style.transform = '';
                        }, 200);
                    }

                    // Vary speed with smooth transitions
                    const speedChange = (Math.random() - 0.5) * 5;
                    v.speed = Math.max(10, Math.min(v.maxSpeed + 10, v.speed + speedChange));

                    // Random events
                    if (Math.random() < 0.05) { // 5% chance to trigger event
                        const events = ['Acel. Brusca', 'Frenagem Brusca', 'Curva Brusca', 'Exc. Velocidade'];
                        v.event = events[Math.floor(Math.random() * events.length)];
                        setTimeout(() => { v.event = null; }, 5000);
                    }
                });

                // Refresh list without resetting scroll
                renderVehicleList(
                    document.getElementById('vehicleSearch').value,
                    currentStatusFilter
                );

                // Update marker positions in popup if open
                if (selectedVehicleId) {
                    const selected = vehicles.find(v => v.id === selectedVehicleId);
                    if (selected) {
                        const popup = document.getElementById('vehiclePopup');
                        if (popup.classList.contains('show')) {
                            showPopup(selected);
                        }
                    }
                }
            }, 2000);
        }

        /* ADD RANDOM WEATHER EFFECT */
        function createWeatherEffect() {
            const mapArea = document.getElementById('mapArea');
            const weatherEl = document.createElement('div');
            weatherEl.style.position = 'absolute';
            weatherEl.style.top = '0';
            weatherEl.style.left = '0';
            weatherEl.style.right = '0';
            weatherEl.style.bottom = '0';
            weatherEl.style.pointerEvents = 'none';
            weatherEl.style.zIndex = '5';
            weatherEl.style.background = 'linear-gradient(180deg, rgba(255,255,255,0) 0%, rgba(255,255,255,0.1) 100%)';
            weatherEl.style.animation = 'fadeInOut 3s ease-in-out';
            mapArea.appendChild(weatherEl);

            setTimeout(() => weatherEl.remove(), 3000);
        }

        /* Trigger random weather every 30 seconds */
        setInterval(createWeatherEffect, 30000);

        /* INIT */
        renderVehicleList();
        renderMarkers();
        updateStats();
        startAnimations();
