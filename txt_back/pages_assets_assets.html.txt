        // Renderizar KPIs
        function renderKPIs() {
            const total = vehicles.length;
            const active = vehicles.filter(v => v.status === 'active').length;
            const indisponivel = vehicles.filter(v => v.status === 'indisponivel').length;
            const desativado = vehicles.filter(v => v.status === 'desativado').length;

            document.getElementById('totalStats').textContent = `${total} ativos`;
            document.getElementById('activeStats').textContent = `${active} ativos`;
            document.getElementById('indisponivelStats').textContent = `${indisponivel} indisponíveis`;

            const kpis = [
                { label: 'Total de Ativos', value: total, icon: 'fa-car-side' },
                { label: 'Ativos', value: active, icon: 'fa-check-circle' },
                { label: 'Indisponíveis', value: indisponivel, icon: 'fa-exclamation-triangle' },
                { label: 'Desativados', value: desativado, icon: 'fa-ban' }
            ];

            const container = document.getElementById('kpiContainer');
            container.innerHTML = kpis.map(kpi => `
                <div class="kpi-card">
                    <div class="kpi-icon">
                        <i class="fas ${kpi.icon}"></i>
                    </div>
                    <div class="kpi-info">
                        <div class="kpi-value">${kpi.value}</div>
                        <div class="kpi-label">${kpi.label}</div>
                    </div>
                </div>
            `).join('');
        }

        // Aplicar filtros
        function applyFilters() {
            const statusRadio = document.querySelector('input[name="statusFilter"]:checked');
            const statusFilter = statusRadio ? statusRadio.value : 'all';
            const typeFilter = document.getElementById('typeFilter').value;
            const yearFilter = document.getElementById('yearFilter').value;
            const searchTerm = document.getElementById('mainSearch').value.toLowerCase();

            filteredVehicles = vehicles.filter(v => {
                // Filtro de status
                if (statusFilter !== 'all' && v.status !== statusFilter) return false;

                // Filtro de tipo
                if (typeFilter !== 'all' && v.type !== typeFilter) return false;

                // Filtro de ano
                if (yearFilter && v.year !== parseInt(yearFilter)) return false;

                // Busca textual
                if (searchTerm) {
                    const matches =
                        v.description.toLowerCase().includes(searchTerm) ||
                        v.plate.toLowerCase().includes(searchTerm) ||
                        v.imei.includes(searchTerm) ||
                        v.model.toLowerCase().includes(searchTerm) ||
                        v.assetId.toLowerCase().includes(searchTerm);
                    if (!matches) return false;
                }

                return true;
            });

            currentPage = 1;
            renderTable();
            renderKPIs();
        }

        // Limpar todos os filtros
        function clearAllFilters() {
            document.querySelector('input[name="statusFilter"][value="all"]').checked = true;
            document.getElementById('typeFilter').value = 'all';
            document.getElementById('yearFilter').value = '';
            document.getElementById('mainSearch').value = '';
            document.getElementById('sidebarSearch').value = '';
            applyFilters();
            showToast('Filtros limpos');
        }

        function loadVehicles() {
            const tbody = document.getElementById('tableBody');
            tbody.innerHTML = Array(5).fill(0).map(() => `

                    ${Array(12).fill().map(() => '<td><div class="skeleton" style="height: 30px; border-radius: 4px;"></div></td>').join('')}

                applyFilters();

        }

        // Renderizar tabela
        function renderTable() {
            const tbody = document.getElementById('tableBody');
            const countSpan = document.getElementById('vehicleCount');

            countSpan.innerText = `${filteredVehicles.length} veículos`;

            if (filteredVehicles.length === 0) {
                tbody.innerHTML = `
                    <tr>
                        <td colspan="12" style="text-align: center; padding: 60px;">
                            <i class="fas fa-car-side" style="font-size: 48px; opacity: 0.3; margin-bottom: 15px;"></i>
                            <p style="color: var(--text-secondary);">Nenhum veículo encontrado</p>
                        </td>
                    </tr>
                `;
                document.getElementById('pagination').innerHTML = '';
                return;
            }

            const start = (currentPage - 1) * itemsPerPage;
            const end = Math.min(start + itemsPerPage, filteredVehicles.length);
            const pageVehicles = filteredVehicles.slice(start, end);

            tbody.innerHTML = pageVehicles.map(v => {
                let statusClass = 'active';
                let statusIcon = 'fa-check-circle';
                let statusText = 'Ativo';

                if (v.status === 'indisponivel') {
                    statusClass = 'indisponivel';
                    statusIcon = 'fa-exclamation-triangle';
                    statusText = 'Indisponível';
                } else if (v.status === 'desativado') {
                    statusClass = 'desativado';
                    statusIcon = 'fa-ban';
                    statusText = 'Desativado';
                }

                return `
                    <tr onclick="showVehicleDetails(${v.id})" style="cursor: pointer;">
                        <td>
                            <span class="status-badge ${statusClass}">
                                <i class="fas ${statusIcon}"></i>
                                ${statusText}
                            </span>
                        </td>
                        <td>
                            <div class="vehicle-info">
                                <i class="fas fa-truck"></i>
                                <div>
                                    <strong>${v.description}</strong>
                                    <small>${v.plate}</small>
                                </div>
                            </div>
                        </td>
                        <td>
                            <div>${v.lastPosition}</div>
                            <small class="text-muted">${v.lastPositionTime}</small>
                        </td>
                        <td>${v.lastTrip}</td>
                        <td>${v.model}</td>
                        <td>${v.registerNumber}</td>
                        <td>${v.garage}</td>
                        <td>${v.year}</td>
                        <td><span class="imei">${v.imei}</span></td>
                        <td>${v.assetId}</td>
                        <td>${v.type}</td>
                        <td>${v.manufacturer}</td>
                    </tr>
                `;
            }).join('');

            renderPagination();
        }

        // Paginação
        function renderPagination() {
            const totalPages = Math.ceil(filteredVehicles.length / itemsPerPage);
            const pagination = document.getElementById('pagination');

            if (totalPages <= 1) {
                pagination.innerHTML = '';
                return;
            }

            let html = `
                <span class="pagination-info">
                    Página ${currentPage} de ${totalPages}
                </span>
                <div class="pagination-controls">
            `;

            html += `<button class="page-btn" ${currentPage === 1 ? 'disabled' : ''} onclick="changePage(${currentPage - 1})"><i class="fas fa-chevron-left"></i></button>`;

            for (let i = 1; i <= totalPages; i++) {
                if (i === 1 || i === totalPages || (i >= currentPage - 2 && i <= currentPage + 2)) {
                    html += `<button class="page-btn ${i === currentPage ? 'active' : ''}" onclick="changePage(${i})">${i}</button>`;
                } else if (i === currentPage - 3 || i === currentPage + 3) {
                    html += `<span class="page-btn" style="border: none;">...</span>`;
                }
            }

            html += `<button class="page-btn" ${currentPage === totalPages ? 'disabled' : ''} onclick="changePage(${currentPage + 1})"><i class="fas fa-chevron-right"></i></button>`;
            html += '</div>';
            pagination.innerHTML = html;
        }

        function changePage(page) {
            currentPage = page;
            renderTable();
        }

        // Exportar CSV
        function exportToCSV() {
            const headers = [
                'Status', 'Descrição', 'Placa', 'Última Posição', 'Última Viagem',
                'Modelo', 'Registro', 'Garagem', 'Ano', 'IMEI', 'ID', 'Tipo', 'Fabricante'
            ];
            const data = filteredVehicles.map(v => [
                v.status === 'active' ? 'Ativo' : v.status === 'indisponivel' ? 'Indisponível' : 'Desativado',
                v.description, v.plate, v.lastPosition, v.lastTrip,
                v.model, v.registerNumber, v.garage, v.year,
                v.imei, v.assetId, v.type, v.manufacturer
            ]);
            const csvContent = [headers.join(','), ...data.map(row => row.map(cell => `"${cell}"`).join(','))].join('\n');
            const blob = new Blob(['\uFEFF' + csvContent], { type: 'text/csv;charset=utf-8;' });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = `ativos_${new Date().toISOString().split('T')[0]}.csv`;
            link.click();
            showToast('Exportação concluída!');
        }

        function showVehicleDetails(id) {
            const vehicle = vehicles.find(v => v.id === id);
            if (vehicle) showToast(`Detalhes de ${vehicle.description}`, 2000);
        }
