        // Estado da aplicação
        let currentTab = 'groups';
        let currentGroupId = 1;
        let currentClientId = 1;

        // Inicialização
        document.addEventListener('DOMContentLoaded', () => {
            // Configurar tema
            const savedTheme = localStorage.getItem('theme') || 'dark';
            document.body.className = `theme-${savedTheme}`;

            loadCompanies();
            loadGroups();
            loadClients();
            setupEventListeners();

            // Selecionar primeiro grupo por padrão
            if (groups.length) selectGroup(groups[0].id);
            if (clients.length) selectClient(clients[0].id);
        });

        // Event listener para mudança de tema
        document.addEventListener('themeChanged', (e) => {
            document.body.className = `theme-${e.detail.theme}`;
        });

        function setupEventListeners() {
            document.querySelectorAll('.config-tab').forEach(tab => {
                tab.addEventListener('click', () => switchTab(tab.dataset.tab));
            });

            document.getElementById('newGroupBtn').addEventListener('click', () => openGroupModal());
            document.getElementById('companySelect').addEventListener('change', (e) => filterByCompany(e.target.value));
            document.getElementById('searchInput').addEventListener('input', (e) => filterList(e.target.value));
            document.getElementById('selectAll').addEventListener('change', (e) => toggleSelectAll(e.target.checked));
        }

        function switchTab(tabName) {
            currentTab = tabName;

            // Atualizar tabs
            document.querySelectorAll('.config-tab').forEach(t => {
                t.classList.toggle('active', t.dataset.tab === tabName);
            });

            // Atualizar conteúdos
            document.querySelectorAll('.tab-content').forEach(c => {
                c.classList.toggle('active', c.id.includes(tabName));
            });

            // Atualizar listas na sidebar
            document.getElementById('groupsList').classList.toggle('active', tabName === 'groups');
            document.getElementById('clientsList').classList.toggle('active', tabName === 'clients');

            // Atualizar placeholder da busca
            const searchInput = document.getElementById('searchInput');
            searchInput.placeholder = tabName === 'groups' ? 'Buscar grupos...' : 'Buscar clientes...';
            searchInput.value = '';
        }

        function loadGroups() {
            const list = document.getElementById('groupsList');
            list.innerHTML = '';

            groups.forEach((g, index) => {
                const div = document.createElement('div');
                div.className = `group-item ${currentGroupId === g.id ? 'active' : ''}`;
                div.style.animationDelay = `${index * 0.1}s`;
                div.innerHTML = `
                    <div class="group-name">
                        <i class="fas fa-layer-group" style="color: var(--brand-orange);"></i>
                        ${g.name}
                    </div>
                    <div class="group-desc">
                        <i class="fas fa-building" style="font-size: 10px;"></i> ${g.company}
                    </div>
                `;
                div.onclick = () => selectGroup(g.id);
                list.appendChild(div);
            });
        }

        function selectGroup(id) {
            currentGroupId = id;
            const group = groups.find(g => g.id === id);

            // Atualizar visual da lista
            loadGroups();

            // Carregar cards de configuração
            const cards = document.getElementById('configCards');
            cards.innerHTML = '';

            const configItems = [
                { icon: 'fa-map-marker-alt', label: 'Rastreamento', value: group.configs.tracking ? 'Ativo' : 'Inativo', status: group.configs.tracking ? 'active' : 'inactive' },
                { icon: 'fa-file-alt', label: 'Relatórios', value: group.configs.reports ? 'Automático' : 'Manual', status: group.configs.reports ? 'active' : 'inactive' },
                { icon: 'fa-bell', label: 'Alertas', value: group.configs.alerts ? 'Ligado' : 'Desligado', status: group.configs.alerts ? 'active' : 'inactive' },
                { icon: 'fa-microchip', label: 'Equipamentos', value: `${group.equipmentCount} unidades`, status: 'active' }
            ];

            configItems.forEach((item, index) => {
                const card = document.createElement('div');
                card.className = 'metric-card';
                card.style.animationDelay = `${index * 0.1}s`;
                card.innerHTML = `
                    <div class="metric-icon">
                        <i class="fas ${item.icon}"></i>
                    </div>
                    <div class="metric-info">
                        <h4>${item.label}</h4>
                        <div class="metric-value">${item.value}</div>
                    </div>
                `;
                cards.appendChild(card);
            });
        }

        function loadClients() {
            const list = document.getElementById('clientsList');
            list.innerHTML = '';

            clients.forEach((c, index) => {
                const div = document.createElement('div');
                div.className = `client-item ${currentClientId === c.id ? 'selected' : ''}`;
                div.style.animationDelay = `${index * 0.1}s`;
                div.innerHTML = `
                    <div class="client-name">
                        <i class="fas fa-building" style="color: var(--brand-orange);"></i>
                        ${c.tradeName}
                    </div>
                    <div class="client-doc">
                        <i class="fas fa-id-card" style="font-size: 10px;"></i> ${c.cnpj}
                    </div>
                `;
                div.onclick = () => selectClient(c.id);
                list.appendChild(div);
            });
        }

        function selectClient(id) {
            currentClientId = id;
            const client = clients.find(c => c.id === id);

            // Atualizar visual da lista
            loadClients();

            // Atualizar info do cliente
            document.getElementById('selectedClientInfo').innerHTML = `
                <i class="fas fa-building"></i>
                Cliente: <strong>${client.tradeName}</strong> (${client.cnpj})
            `;

            // Carregar métricas do cliente
            const metrics = document.getElementById('clientMetrics');
            metrics.innerHTML = '';

            const clientMetrics = [
                { icon: 'fa-crown', label: 'Plano', value: client.plan.charAt(0).toUpperCase() + client.plan.slice(1) },
                { icon: 'fa-chart-line', label: 'Consumo', value: `${client.dataConsumption} MB` },
                { icon: 'fa-dollar-sign', label: 'Valor', value: `R$ ${client.planValue.toFixed(2)}` },
                { icon: 'fa-microchip', label: 'Equipamentos', value: `${client.equipmentCount} unidades` }
            ];

            clientMetrics.forEach((item, index) => {
                const card = document.createElement('div');
                card.className = 'metric-card';
                card.style.animationDelay = `${index * 0.1}s`;
                card.innerHTML = `
                    <div class="metric-icon">
                        <i class="fas ${item.icon}"></i>
                    </div>
                    <div class="metric-info">
                        <h4>${item.label}</h4>
                        <div class="metric-value">${item.value}</div>
                    </div>
                `;
                metrics.appendChild(card);
            });

            // Carregar equipamentos
            loadEquipments();
        }

        function loadEquipments() {
            const tbody = document.getElementById('equipmentTableBody');
            tbody.innerHTML = '';

            equipments.forEach((e, index) => {
                const row = document.createElement('tr');
                row.style.animationDelay = `${index * 0.1}s`;
                row.innerHTML = `
                    <td><input type="checkbox" class="equipment-check"></td>
                    <td><strong>${e.imei}</strong></td>
                    <td>${e.vehicle}</td>
                    <td>${e.plate}</td>
                    <td>${e.model}</td>
                    <td><span class="status-badge ${e.status}">${e.status}</span></td>
                    <td>${e.lastCom}</td>
                    <td>
                        <button class="btn-icon" onclick="showToast('Editando equipamento')">
                            <i class="fas fa-edit"></i>
                        </button>
                        <button class="btn-icon" onclick="showToast('Excluindo equipamento')">
                            <i class="fas fa-trash"></i>
                        </button>
                    </td>
                `;
                tbody.appendChild(row);
            });
        }

        function loadCompanies() {
            const select = document.getElementById('companySelect');
            const groupCompanySelect = document.getElementById('groupCompany');

            const options = clients.map(c => `<option value="${c.id}">${c.tradeName}</option>`).join('');
            select.innerHTML = '<option value="">Todas as Empresas</option>' + options;
            groupCompanySelect.innerHTML = options;
        }

        function filterByCompany(id) {
            if (id) {
                switchTab('clients');
                selectClient(parseInt(id));
            }
        }

        function filterList(query) {
            const items = currentTab === 'groups'
                ? document.querySelectorAll('.group-item')
                : document.querySelectorAll('.client-item');

            items.forEach(item => {
                const text = item.textContent.toLowerCase();
                if (text.includes(query.toLowerCase())) {
                    item.style.display = 'block';
                } else {
                    item.style.display = 'none';
                }
            });
        }

        function toggleSelectAll(checked) {
            document.querySelectorAll('.equipment-check').forEach(cb => {
                cb.checked = checked;
            });
        }

        function openGroupModal() {
            document.getElementById('groupModal').classList.add('open');
            document.getElementById('groupName').value = '';
            document.getElementById('groupDesc').value = '';
            document.getElementById('configTracking').checked = true;
            document.getElementById('configReports').checked = true;
            document.getElementById('configAlerts').checked = true;
        }

        function saveGroup() {
            const name = document.getElementById('groupName').value;
            if (!name) {
                showToast('Nome do grupo é obrigatório', 'error');
                return;
            }

            const company = document.getElementById('groupCompany').value;
            const desc = document.getElementById('groupDesc').value;

            groups.push({
                id: groups.length + 1,
                name: name,
                description: desc,
                company: company,
                configs: {
                    tracking: document.getElementById('configTracking').checked,
                    reports: document.getElementById('configReports').checked,
                    alerts: document.getElementById('configAlerts').checked
                },
                equipmentCount: 0
            });

            loadGroups();
            closeModal('groupModal');
            showToast('Grupo criado com sucesso!');
        }

        function closeModal(id) {
            document.getElementById(id).classList.remove('open');
        }

        function showToast(message, type = 'success') {
            const toast = document.getElementById('toast');
            const icon = toast.querySelector('i');

            if (type === 'success') {
                icon.className = 'fas fa-check-circle';
                toast.style.background = 'var(--brand-orange)';
            } else {
                icon.className = 'fas fa-exclamation-circle';
                toast.style.background = '#e74c3c';
            }

            document.getElementById('toastMessage').textContent = message;
            toast.style.display = 'flex';

            setTimeout(() => {
                toast.style.display = 'none';
            }, 3000);
        }

        // Adicionar listeners para botões
        document.getElementById('importEquipmentsBtn').addEventListener('click', () => {
            showToast('Função de importação em desenvolvimento');
        });

        document.getElementById('addEquipmentBtn').addEventListener('click', () => {
            showToast('Função de adicionar equipamento em desenvolvimento');
        });
