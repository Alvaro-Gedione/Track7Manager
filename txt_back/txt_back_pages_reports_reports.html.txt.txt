    <script>
        // Adicionar no topo do <script>
        const API = 'http://localhost:5000/api';
        let _unitsCache = [];   // [{id, name}]
        let _selectedUnitIds = new Set(); // persiste entre steps
        let _reportTimeFrom = 0;
        let _reportTimeTo = 0;
        let _selectedPeriodLabel = '';
        let _reportTitle = '';

        async function loadUnitsForModal() {
            if (_unitsCache.length > 0) return _unitsCache;
            try {
                const r = await fetch(`${API}/units`);
                const d = await r.json();
                if (d.error) throw new Error(d.error);
                _unitsCache = d.units; // [{id, name}]
                return _unitsCache;
            } catch (e) {
                showToast('Erro ao carregar unidades: ' + e.message);
                return [];
            }
        }
        // Configurar tema inicial baseado no localStorage
        document.addEventListener('DOMContentLoaded', async () => {
            const savedTheme = localStorage.getItem('theme') || 'dark';
            document.body.className = `theme-${savedTheme}`;

            // Garante sessão ativa antes de qualquer coisa
            try {
                const st = await fetch(`${API}/status`);
                const sd = await st.json();
                if (!sd.authenticated) {
                    await fetch(`${API}/login`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ token: '67eba42da80819eb285dd8794678964707481B6FFB3E065405FE46F889143875DE5B3C3A' })
                    });
                }
                await loadUnitsForModal(); // pré-carrega o cache
            } catch (e) {
                console.warn('Servidor offline:', e.message);
            }

            animateStats();
        });

        // Ouvir mudanças de tema da navegação
        document.addEventListener('themeChanged', (e) => {
            document.body.className = `theme-${e.detail.theme}`;
        });

        // Variáveis do modal
        let currentStep = 1;
        let reportTitle = '';

        /* ── ABRIR / FECHAR MODAL ── */
        function openModal(title) {
            _reportTitle = title;
            _selectedUnitIds = new Set(); // limpa seleções anteriores
            _reportTimeFrom = 0;
            _reportTimeTo = 0;
            _selectedPeriodLabel = '';
            currentStep = 1;

            document.getElementById('modalTitle').querySelector('span').textContent = title;
            renderStep();
            document.getElementById('modalOverlay').classList.add('open');
        }

        // renderStep precisa chamar renderStep1 de forma async
        function renderStep() {
            ['s1', 's2', 's3'].forEach((id, i) => {
                const el = document.getElementById(id);
                el.className = (i + 1 === currentStep) ? 'step current'
                    : (i + 1 < currentStep) ? 'step done' : 'step';
            });
            const back = document.getElementById('btnBack');
            const next = document.getElementById('btnNext');
            back.style.display = currentStep === 1 ? 'none' : 'inline-block';

            if (currentStep === 3) {
                next.innerHTML = '▶ Rodar <i class="fas fa-play"></i>';
                next.classList.add('btn-run');
            } else {
                next.innerHTML = 'Próximo <i class="fas fa-arrow-right"></i>';
                next.classList.remove('btn-run');
            }

            const body = document.getElementById('modalBody');
            if (currentStep === 1) renderStep1(body);      // async, tudo bem
            if (currentStep === 2) renderStep2(body);
            if (currentStep === 3) renderStep3(body);
        }

        // nextStep: quando é step 3, delega ao runReport async
        function nextStep() {
            if (currentStep === 1) {
                // Valida pelo Set global, não pelo DOM
                if (_selectedUnitIds.size === 0) {
                    showToast('⚠️ Selecione pelo menos uma unidade.');
                    return;
                }
            }

            if (currentStep === 2 && !_reportTimeFrom) {
                [_reportTimeFrom, _reportTimeTo] = calcPeriod('Últimos 30 dias');
            }

            if (currentStep < 3) {
                currentStep++;
                renderStep();
            } else {
                runReport();
            }
        }

        function closeModal() {
            document.getElementById('modalOverlay').classList.remove('open');
        }

        document.getElementById('modalOverlay').addEventListener('click', function (e) {
            if (e.target === this) closeModal();
        });

        function prevStep() {
            if (currentStep > 1) {
                currentStep--;
                renderStep();
                showToast(`Passo ${currentStep} de 3`);
            }
        }


        /* ── STEP 1: ATIVOS ── */
        async function renderStep1(body) {
            // Loading enquanto busca
            body.innerHTML = `
        <div style="padding:30px;text-align:center;color:var(--brand-silver);">
            <div class="spinner" style="margin:0 auto 10px;"></div>
            Carregando unidades...
        </div>`;

            if (!_unitsCache.length) {
                try {
                    const r = await fetch(`${API}/units`);
                    const d = await r.json();
                    if (d.error) throw new Error(d.error);

        function calcPeriod(label) {
            const now = Math.floor(Date.now() / 1000);
            const d = new Date(); d.setHours(0, 0, 0, 0);
            const today = Math.floor(d.getTime() / 1000);
            const yrStart = Math.floor(new Date(d.getFullYear(), 0, 1).getTime() / 1000);

            const map = {
                'Hoje': [today, now],
                'Ontem': [today - 86400, today - 1],
                'Ontem e Hoje': [today - 86400, now],
                'Semana Corrente': [today - (d.getDay() * 86400), now],
                'Últimos 7 dias': [now - 7 * 86400, now],
                'Últimos 7 dias (sem hoje)': [now - 7 * 86400, today - 1],
                'Mês Corrente': [Math.floor(new Date(d.getFullYear(), d.getMonth(), 1).getTime() / 1000), now],
                'Últimos 30 dias': [now - 30 * 86400, now],
                'Últimos 30 dias (sem hoje)': [now - 30 * 86400, today - 1],
                'Último Mês': [Math.floor(new Date(d.getFullYear(), d.getMonth() - 1, 1).getTime() / 1000),
                Math.floor(new Date(d.getFullYear(), d.getMonth(), 0, 23, 59, 59).getTime() / 1000)],
                'Acumulado do Ano': [yrStart, now],
                'Últimos 2 Meses': [now - 60 * 86400, now],
                'Últimos 6 Meses': [now - 180 * 86400, now],
            };
            return map[label] || [now - 30 * 86400, now];
        }

        /* ── RODAR RELATÓRIO ── */
        async function runReport() {
            const units = getSelectedUnits();
            if (!units.length) {
                showToast('⚠️ Selecione pelo menos uma unidade.');
                return;
            }

            // Garante que temos período
            if (!_reportTimeFrom) [_reportTimeFrom, _reportTimeTo] = calcPeriod('Últimos 30 dias');

            // UI de loading
            const progressBar = document.getElementById('progressBar');
            const progressFill = document.getElementById('progressFill');
            progressBar.classList.add('show');
            progressFill.style.width = '0%';
            document.getElementById('btnNext').disabled = true;

            let allMsgs = [];
            let allParamKeys = new Set();

            for (let i = 0; i < units.length; i++) {
                const u = units[i];
                const pct = Math.round((i / units.length) * 100);
                progressFill.style.width = pct + '%';
                showToast(`Buscando ${i + 1}/${units.length}: ${u.name}`);

                try {
                    const r = await fetch(`${API}/messages`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            unitId: u.id,
                            timeFrom: _reportTimeFrom,
                            timeTo: _reportTimeTo,
                            msgType: 1,
                            maxCount: 5000
                        })
                    });
                    const d = await r.json();
                    if (d.error) throw new Error(d.error);

                    (d.messages || []).forEach(m => {
                        m._unitName = u.name;
                        m._unitId = u.id;
                        Object.keys(m.p || {}).forEach(k => allParamKeys.add(k));
                        allMsgs.push(m);
                    });
                } catch (e) {
                    showToast(`Erro em "${u.name}": ${e.message}`);
                }
            }

            progressFill.style.width = '100%';
            setTimeout(() => progressBar.classList.remove('show'), 600);
            document.getElementById('btnNext').disabled = false;

            closeModal();
            renderReportTable(allMsgs, allParamKeys);
            showToast(`✅ ${allMsgs.length.toLocaleString('pt-BR')} mensagens carregadas para "${_reportTitle}"`);
        }

        function renderReportTable(msgs, paramKeys) {
            const paramArr = [...paramKeys].sort();

            // Mesmas colunas do list.html
            const cols = [
                { key: '_idx', label: '#', mono: true, unitCol: false },
                { key: 'unit', label: 'Unidade', mono: false, unitCol: true },
                { key: 'dt', label: 'Data / Hora', mono: true, unitCol: false },
                { key: 'lat', label: 'Latitude', mono: true, unitCol: false },
                { key: 'lon', label: 'Longitude', mono: true, unitCol: false },
                { key: 'spd', label: 'Vel.(km/h)', speed: true, unitCol: false },
                { key: 'course', label: 'Curso(°)', mono: false, unitCol: false },
                { key: 'alt', label: 'Alt.(m)', mono: false, unitCol: false },
                { key: 'sat', label: 'Sats', sat: true, unitCol: false },
            ];
            paramArr.forEach(k => cols.push({ key: 'p_' + k, label: k, param: true }));
            rptColHeaders = cols;

            const rows = [];
            msgs.forEach((m, idx) => {
                const pos = m.pos || {};
                const row = {
                    _idx: idx + 1,
                    unit: m._unitName || '',
                    t: m.t || 0,
                    dt: tsToStr(m.t),
                    lat: pos.y != null ? pos.y.toFixed(6) : '',
                    lon: pos.x != null ? pos.x.toFixed(6) : '',
                    spd: pos.s != null ? pos.s : '',
                    course: pos.c != null ? pos.c : '',
                    alt: pos.z != null ? pos.z : '',
                    sat: pos.sc != null ? pos.sc : '',
                };
                const p = m.p || {};
                paramArr.forEach(k => {
                    const v = p[k];
                    row['p_' + k] = (v != null && typeof v === 'object')
                        ? (v.v != null ? v.v : JSON.stringify(v))
                        : (v != null ? v : '');
                });
                rows.push(row);
            });

            // Stats
            const speeds = rows.map(r => parseFloat(r.spd)).filter(v => !isNaN(v) && v >= 0);
            const unitSet = new Set(rows.map(r => r.unit));
            document.getElementById('rptTotal').textContent = rows.length.toLocaleString('pt-BR');
            document.getElementById('rptUnits').textContent = unitSet.size;
            document.getElementById('rptMaxSpd').textContent = (speeds.length ? Math.max(...speeds) : 0) + ' km/h';
            document.getElementById('rptAvgSpd').textContent = (speeds.length
                ? (speeds.reduce((a, b) => a + b, 0) / speeds.length).toFixed(1) : 0) + ' km/h';
            document.getElementById('rptPeriod').textContent = _selectedPeriodLabel || '--';

            rptAllRows = rows;
            rptFiltered = [...rows];
            rptPage = 1;

            document.getElementById('reportResultArea').style.display = 'block';
            document.getElementById('reportGrid').style.display = 'none'; // oculta os cards
            renderRptTable();
            document.getElementById('reportResultArea').scrollIntoView({ behavior: 'smooth' });
        }

        function renderRptTable() {
            const start = (rptPage - 1) * RPT_PAGE_SIZE;
            const page = rptFiltered.slice(start, start + RPT_PAGE_SIZE);

            // Header
            document.getElementById('rptTHead').innerHTML = '<tr>' + rptColHeaders.map(c =>
                `<th onclick="rptSortBy('${c.key}')" style="
            position:sticky;top:0;background:var(--table-header-bg,#181818);
            border-bottom:2px solid var(--border-color);padding:8px 10px;
            text-align:left;font-size:11px;font-weight:700;
            color:${c.unitCol ? 'var(--brand-orange)' : 'var(--brand-silver)'};
            text-transform:uppercase;letter-spacing:.5px;white-space:nowrap;cursor:pointer;
        ">${c.label} <span style="opacity:.4;font-size:9px;">
            ${rptSortCol === c.key ? (rptSortAsc ? '▲' : '▼') : '⇅'}
        </span></th>`
            ).join('') + '</tr>';

            // Body
            document.getElementById('rptTBody').innerHTML = page.map(row => {
                const spd = parseFloat(row.spd);
                const hasSpd = !isNaN(spd);
                const spdColor = spd > 90 ? '#e53e3e' : spd > 60 ? '#e67e22' : '#27ae60';
                const spdCls = spd > 90 ? 'bg-red' : spd > 60 ? 'bg-orange' : 'bg-green';
                const spdW = hasSpd ? Math.min(spd / 120 * 36, 36) : 0;

                const cells = rptColHeaders.map(c => {
                    const v = row[c.key];
                    const str = v != null && v !== '' ? String(v) : '';

                    if (c.key === '_idx')
                        return `<td style="color:#666;font-size:11px;padding:6px 10px;">${v}</td>`;
                    if (c.unitCol)
                        return `<td style="font-weight:600;color:var(--brand-orange);padding:6px 10px;
                         max-width:180px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;"
                         title="${str}">${str}</td>`;
                    if (c.speed && hasSpd)
                        return `<td style="padding:6px 10px;">
                    <span style="display:inline-block;height:6px;border-radius:3px;
                        width:${spdW}px;background:${spdColor};vertical-align:middle;margin-right:5px;"></span>
                    <span style="background:${spdColor}22;color:${spdColor};
                        padding:1px 6px;border-radius:3px;font-size:10.5px;font-weight:700;">${spd} km/h</span></td>`;
                    if (c.sat && str)
                        return `<td style="padding:6px 10px;">
                    <span style="background:${parseInt(str) >= 4 ? 'rgba(46,125,50,.3)' : 'rgba(183,28,28,.3)'};
                        color:${parseInt(str) >= 4 ? '#81c784' : '#ff8a80'};
                        padding:1px 6px;border-radius:3px;font-size:10.5px;font-weight:700;">${str}</span></td>`;
                    if (c.param)
                        return str
                            ? `<td style="padding:6px 10px;"><span style="background:rgba(242,101,34,.15);
                        border:1px solid rgba(242,101,34,.3);border-radius:3px;padding:1px 5px;
                        font-size:10.5px;color:#ff8c52;font-family:monospace;">${str}</span></td>`
                            : `<td style="padding:6px 10px;color:#444;">-</td>`;
                    if (c.mono)
                        return `<td style="padding:6px 10px;font-family:monospace;font-size:11.5px;
                    color:var(--brand-silver);">${str || '<span style="color:#444;">-</span>'}</td>`;
                    return `<td style="padding:6px 10px;">${str || '<span style="color:#444;">-</span>'}</td>`;
                });

                return `<tr style="border-bottom:1px solid var(--border-color);">${cells.join('')}</tr>`;
            }).join('');

            renderRptPagination();
        }

        function renderRptPagination() {
            const total = rptFiltered.length;
            const totalPages = Math.ceil(total / RPT_PAGE_SIZE);
            const start = (rptPage - 1) * RPT_PAGE_SIZE + 1;
            const end = Math.min(rptPage * RPT_PAGE_SIZE, total);

            document.getElementById('rptPagInfo').textContent =
                `Exibindo ${start.toLocaleString('pt-BR')}-${end.toLocaleString('pt-BR')} de ${total.toLocaleString('pt-BR')}`;
            document.getElementById('rptPageInfo').textContent =
                `Pág. ${rptPage}/${totalPages}`;

            const btns = document.getElementById('rptPagBtns');
            btns.innerHTML = '';
            const addBtn = (label, pg, active) => {
                const b = document.createElement('button');
                b.innerHTML = label;
                b.disabled = pg < 1 || pg > totalPages || active;
                b.style.cssText = `padding:4px 10px;border:1px solid var(--border-color);
            border-radius:3px;background:${active ? 'var(--brand-orange)' : 'var(--input-bg)'};
            color:${active ? '#fff' : 'var(--text-primary)'};font-size:12px;cursor:pointer;`;
                b.onclick = () => { rptPage = pg; renderRptTable(); };
                btns.appendChild(b);
            };
            addBtn('◀ Ant', rptPage - 1, false);
            const pages = new Set([1, totalPages, rptPage, rptPage - 1, rptPage + 1]
                .filter(p => p >= 1 && p <= totalPages));
            let last = 0;
            [...pages].sort((a, b) => a - b).forEach(p => {
                if (last && p - last > 1) {
                    const sp = document.createElement('span');
                    sp.textContent = '...'; sp.style.cssText = 'padding:4px;color:#666;';
                    btns.appendChild(sp);
                }
                addBtn(p, p, p === rptPage);
                last = p;
            });
            addBtn('Próx ▶', rptPage + 1, false);
        }

        function rptSortBy(col) {
            rptSortAsc = rptSortCol === col ? !rptSortAsc : true;
            rptSortCol = col;
            rptFiltered.sort((a, b) => {
                let va = a[col], vb = b[col];
                if (!isNaN(parseFloat(va)) && !isNaN(parseFloat(vb))) { va = parseFloat(va); vb = parseFloat(vb); }
                if (va < vb) return rptSortAsc ? -1 : 1;
                if (va > vb) return rptSortAsc ? 1 : -1;
                return 0;
            });
            rptPage = 1;
            renderRptTable();
        }

        function applyRptSearch() {
            const q = document.getElementById('rptSearch').value.toLowerCase();
            rptFiltered = !q ? [...rptAllRows] : rptAllRows.filter(row =>
                rptColHeaders.some(c => {
                    const v = row[c.key];
                    return v != null && String(v).toLowerCase().includes(q);
                })
            );
            rptPage = 1;
            renderRptTable();
        }

        function exportRptCSV() {
            if (!rptFiltered.length) { showToast('Sem dados.'); return; }
            const hdr = rptColHeaders.map(c => `"${c.label}"`).join(',');
            const rows = rptFiltered.map(r => rptColHeaders.map(c => {
                const v = r[c.key];
                return v != null && v !== '' ? `"${String(v).replace(/"/g, '""')}"` : '""';
            }).join(','));
            const blob = 'data:text/csv;charset=utf-8,' + encodeURIComponent([hdr, ...rows].join('\n'));
            const a = document.createElement('a');
            a.href = blob; a.download = 'relatorio_track7.csv'; a.click();
            showToast(`CSV exportado — ${rptFiltered.length.toLocaleString('pt-BR')} linhas.`);
        }

        }

        function tsToStr(ts) {
            if (!ts) return '';
            const d = new Date(ts * 1000);
            const p = n => String(n).padStart(2, '0');
            return `${p(d.getDate())}/${p(d.getMonth() + 1)}/${d.getFullYear()} ${p(d.getHours())}:${p(d.getMinutes())}:${p(d.getSeconds())}`;
        }
