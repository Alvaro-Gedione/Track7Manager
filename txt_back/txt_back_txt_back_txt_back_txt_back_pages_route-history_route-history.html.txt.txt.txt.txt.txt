        let theme = localStorage.getItem('theme') || 'dark';
        let playbackMultiplier = 1; // Multiplicador de velocidade
        let eventLayers; // Grupo para limpar ícones de eventos facilmente

        // Inicialização
        document.addEventListener('DOMContentLoaded', async () => {
            document.body.className = `theme-${theme}`;

            // Ouvir mudanças de tema
            document.addEventListener('themeChanged', (e) => {
                theme = e.detail.theme;
                document.body.className = `theme-${theme}`;
                updateMapTheme();
            });

            initMap();
            await loadVehicles();
            setDefaultDates();
        });

        function initMap() {
            const mapConfig = {
                zoomControl: true,
                attributionControl: true
            };

            map = L.map('historyMap', mapConfig).setView([-22.2568, -45.7035], 13);

            // Configurar camada base baseada no tema
            updateMapTheme();

            // Adicionar escala
            L.control.scale({ imperial: false, metric: true }).addTo(map);

            // Inicializar layer group de eventos
            eventLayers = L.layerGroup().addTo(map);
        }

        function updateMapTheme() {
            // Remover camadas existentes
            map.eachLayer((layer) => {
                if (layer instanceof L.TileLayer) {
                    map.removeLayer(layer);
                }
            });

            // Adicionar camada apropriada para o tema
            if (theme === 'light') {
                L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', {
                    attribution: '© OpenStreetMap, © CartoDB'
                }).addTo(map);
            } else {
                L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
                    attribution: '© OpenStreetMap, © CartoDB'
                }).addTo(map);
            }
        }

        async function loadVehicles() {
            try {
                // Autenticar primeiro

                const response = await fetch(`${API}/units`);

                if (data.units) {
                    vehicles = data.units;

                    vehicles.forEach(vehicle => {
                        const option = document.createElement('option');
                        option.value = vehicle.id;
                        option.textContent = `${vehicle.name} - ${vehicle.label || 'SEM PLACA'}`;
                        selector.appendChild(option);
                    });
                }
            } catch (error) {
                console.error('Erro ao carregar veículos:', error);
                loadSimulatedVehicles();
            }
        }

        function setDefaultDates() {
            const today = new Date();
            const yesterday = new Date(today);
            yesterday.setDate(yesterday.getDate() - 1);

            document.getElementById('startDate').value = yesterday.toISOString().split('T')[0];
            document.getElementById('endDate').value = today.toISOString().split('T')[0];
        }

        function setTimePreset(preset) {
            const endDate = new Date();
            const startDate = new Date();

            switch (preset) {
                case 'today':
                    startDate.setHours(0, 0, 0, 0);
                    endDate.setHours(23, 59, 59, 999);
                    break;
                case 'yesterday':
                    startDate.setDate(startDate.getDate() - 1);
                    startDate.setHours(0, 0, 0, 0);
                    endDate.setDate(endDate.getDate() - 1);
                    endDate.setHours(23, 59, 59, 999);
                    break;
                case 'week':
                    startDate.setDate(startDate.getDate() - 7);
                    break;
                case 'month':
                    startDate.setMonth(startDate.getMonth() - 1);
                    break;
            }

            document.getElementById('startDate').value = startDate.toISOString().split('T')[0];
            document.getElementById('endDate').value = endDate.toISOString().split('T')[0];

            loadRouteHistory();
        }

        function setPlaybackMultiplier(multiplier, btn) {
            playbackMultiplier = multiplier;
            // UI Update
            document.querySelectorAll('.speed-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');

            // Se estiver tocando, reinicia o intervalo com a nova velocidade
            if (isPlaying) {
                playbackPause();
                playbackStart();
            }
        }

        async function loadRouteHistory() {
            const vehicleId = document.getElementById('vehicleSelector').value;
            const dateStart = document.getElementById('startDate').value;
            const dateEnd = document.getElementById('endDate').value;

            if (!vehicleId) {
                showToast('Selecione um veículo');
                return;
            }

            showToast('Buscando telemetria...', 2000);

            try {
                // Converte as datas dos inputs para timestamps Unix (segundos)
                const timeFrom = Math.floor(new Date(dateStart + "T00:00:00").getTime() / 1000);
                const timeTo = Math.floor(new Date(dateEnd + "T23:59:59").getTime() / 1000);

                const response = await fetch(`${API}/messages`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        unitId: parseInt(vehicleId),
                        timeFrom: timeFrom,
                        timeTo: timeTo,
                        msgType: 1, // Mensagens de posição
                        maxCount: 1000 // Limite para não travar o browser
                    })
                });

                const data = await response.json();

                if (data.messages && data.messages.length > 0) {
                    const formattedRoute = processMessages(data.messages);
                    displayRoute(formattedRoute);

                    document.getElementById('playBtn').disabled = false;
                    document.getElementById('timelineSlider').disabled = false;
                    updateMapStatus(`Rota carregada: ${data.messages.length} pontos`);
                } else {
                    showToast('Nenhum dado encontrado para este período.');
                }

            } catch (error) {
                console.error('Erro na requisição:', error);
                showToast('❌ Erro ao carregar dados do servidor.');
            }
        }

        // Função auxiliar para transformar mensagens brutas em objeto de rota
        function processMessages(messages) {
            const points = messages.map(msg => ({
                lat: msg.pos.y,
                lng: msg.pos.x,
                speed: msg.pos.s || 0,
                timestamp: new Date(msg.t * 1000).toISOString()
            }));

            // Cálculos básicos (estimativas)
            const maxSpeed = Math.max(...points.map(p => p.speed));
            const avgSpeed = points.reduce((acc, p) => acc + p.speed, 0) / points.length;

            return {
                points: points,
                startTime: points[0].timestamp,
                endTime: points[points.length - 1].timestamp,
                totalDistance: (points.length * 0.15).toFixed(1), // Estimativa baseada em pontos
                avgSpeed: Math.round(avgSpeed),
                maxSpeed: Math.round(maxSpeed),
                stops: points.filter(p => p.speed === 0).length,
                fuelConsumption: "--"
            };
        }

        async function fetchRouteData(vehicleId) {
            // Implementar chamada real à API quando disponível
            // Por enquanto, retorna dados simulados
            return generateSimulatedRoute();
        }

        function generateSimulatedRoute() {
            const points = [];
            const centerLat = -22.2568;
            const centerLng = -45.7035;

            // Gerar uma rota circular
            for (let i = 0; i < 50; i++) {
                const angle = (i / 50) * Math.PI * 2;
                const radius = 0.02;

                points.push({
                    lat: centerLat + Math.sin(angle) * radius,
                    lng: centerLng + Math.cos(angle) * radius,
                    speed: 30 + Math.sin(angle * 4) * 20,
                    timestamp: new Date(Date.now() - (50 - i) * 60000).toISOString()
                });
            }

            return {
                points: points,
                startTime: points[0].timestamp,
                endTime: points[points.length - 1].timestamp,
                totalDistance: 45.3,
                avgSpeed: 42,
                maxSpeed: 87,
                stops: 3,
                fuelConsumption: 12.5
            };
        }

        function displayRoute(routeData) {
            // Limpar camadas anteriores
            if (routeLayer) map.removeLayer(routeLayer);
            if (startMarker) map.removeLayer(startMarker);
            if (endMarker) map.removeLayer(endMarker);
            if (currentMarker) map.removeLayer(currentMarker);
            if (eventLayers) eventLayers.clearLayers(); // Limpa eventos anteriores

            currentRoute = routeData.points;
            waypoints = routeData.points;

            // Criar linha da rota
            const latlngs = routeData.points.map(p => [p.lat, p.lng]);

            // Determinar cor da linha baseada na velocidade
            routeLayer = L.polyline(latlngs, {
                color: theme === 'light' ? '#BE6004' : '#BE6004',
                weight: 4,
                opacity: 0.8,
                lineJoin: 'round'
            }).addTo(map);

            // Adicionar marcadores de velocidade ao longo da rota
            routeData.points.forEach((point, index) => {
                if (index % 5 === 0) { // A cada 5 pontos
                    const speedColor = point.speed > 80 ? '#e74c3c' : '#2ecc71';
                    L.circleMarker([point.lat, point.lng], {
                        radius: 3,
                        color: speedColor,
                        fillColor: speedColor,
                        fillOpacity: 0.8
                    }).addTo(map);
                }
            });

            // Adicionar marcadores de eventos
            routeData.points.forEach((point, index) => {
                // Exemplo: Ícone de Excesso de Velocidade (> 80km/h)
                if (point.speed > 80) {
                    L.marker([point.lat, point.lng], {
                        icon: L.divIcon({
                            className: 'event-marker-icon event-speeding',
                            html: '<i class="fas fa-bolt"></i>',
                            iconSize: [20, 20],
                            iconAnchor: [10, 10]
                        })
                    }).bindTooltip(`Excesso de Velocidade: ${Math.round(point.speed)} km/h`).addTo(eventLayers);
                }

                // Exemplo: Ícone de Parada (Velocidade 0 por algum tempo)
                // Aqui você pode melhorar a lógica checando o tempo parado
                if (point.speed === 0 && index % 10 === 0) {
                    L.marker([point.lat, point.lng], {
                        icon: L.divIcon({
                            className: 'event-marker-icon event-stop',
                            html: '<i class="fas fa-parking"></i>',
                            iconSize: [20, 20],
                            iconAnchor: [10, 10]
                        })
                    }).bindTooltip('Ponto de Parada').addTo(eventLayers);
                }
            });

            // Marcador de início
            startMarker = L.marker([routeData.points[0].lat, routeData.points[0].lng], {
                icon: L.divIcon({
                    className: 'custom-div-icon',
                    html: '<div class="start-marker"></div>',
                    iconSize: [16, 16],
                    iconAnchor: [8, 8]
                })
            }).addTo(map).bindTooltip('Início da rota', { permanent: false });

            // Marcador de fim
            endMarker = L.marker([routeData.points[routeData.points.length - 1].lat, routeData.points[routeData.points.length - 1].lng], {
                icon: L.divIcon({
                    className: 'custom-div-icon',
                    html: '<div class="end-marker"></div>',
                    iconSize: [16, 16],
                    iconAnchor: [8, 8]
                })
            }).addTo(map).bindTooltip('Fim da rota', { permanent: false });

            // Marcador da posição atual (início)
            currentMarker = L.marker([routeData.points[0].lat, routeData.points[0].lng], {
                icon: L.divIcon({
                    className: 'custom-div-icon',
                    html: '<div class="vehicle-position-marker"></div>',
                    iconSize: [20, 20],
                    iconAnchor: [10, 10]
                })
            }).addTo(map);

            // Ajustar zoom para mostrar toda a rota
            map.fitBounds(routeLayer.getBounds(), { padding: [50, 50] });

            // Atualizar timeline
            updateTimelineRange();

            // Atualizar informações
            document.getElementById('totalDistance').textContent = routeData.totalDistance + ' km';
            document.getElementById('avgSpeed').textContent = routeData.avgSpeed + ' km/h';
            document.getElementById('maxSpeed').textContent = routeData.maxSpeed + ' km/h';
            document.getElementById('stops').textContent = routeData.stops;
            document.getElementById('fuelConsumption').textContent = routeData.fuelConsumption + ' L';

            // Calcular duração
            const duration = calculateDuration(routeData.startTime, routeData.endTime);
            document.getElementById('totalDuration').textContent = duration;

            // Atualizar labels de tempo
            document.getElementById('timeStart').textContent = formatTime(routeData.startTime);
            document.getElementById('timeEnd').textContent = formatTime(routeData.endTime);
            document.getElementById('timeCurrent').textContent = formatTime(routeData.startTime);

            // Mostrar speed indicator
            document.getElementById('speedIndicator').style.display = 'block';
            updateCurrentSpeed(routeData.points[0].speed);
        }

        function updateTimelineRange() {
            const slider = document.getElementById('timelineSlider');
            slider.max = waypoints.length - 1;
            slider.value = 0;
        }

        function timelineUpdate(value) {
            if (!waypoints.length) return;

            const index = parseInt(value);
            currentWaypointIndex = index;

            const waypoint = waypoints[index];
            if (!waypoint) return;

            // Atualizar posição do marcador
            currentMarker.setLatLng([waypoint.lat, waypoint.lng]);

            // Atualizar tempo atual
            document.getElementById('timeCurrent').textContent = formatTime(waypoint.timestamp);

            // Atualizar velocidade
            updateCurrentSpeed(waypoint.speed);

            // Mover mapa suavemente
            map.panTo([waypoint.lat, waypoint.lng]);
        }

        function updateCurrentSpeed(speed) {
            document.getElementById('currentSpeed').textContent = Math.round(speed);

            // Mudar cor baseada na velocidade
            const speedElement = document.getElementById('currentSpeed');
            if (speed > 80) {
                speedElement.style.color = '#e74c3c';
            } else {
                speedElement.style.color = 'var(--brand-orange)';
            }
        }

        function formatTime(timestamp) {
            const date = new Date(timestamp);
            return date.toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit', second: '2-digit' });
        }

        function calculateDuration(start, end) {
            const startDate = new Date(start);
            const endDate = new Date(end);
            const diffMs = endDate - startDate;
            const diffHrs = Math.floor(diffMs / 3600000);
            const diffMins = Math.floor((diffMs % 3600000) / 60000);
            const diffSecs = Math.floor((diffMs % 60000) / 1000);

            return `${String(diffHrs).padStart(2, '0')}:${String(diffMins).padStart(2, '0')}:${String(diffSecs).padStart(2, '0')}`;
        }

        function playbackStart() {
            if (!waypoints.length || currentWaypointIndex >= waypoints.length - 1) {
                currentWaypointIndex = 0;
            }

            isPlaying = true;
            document.getElementById('playBtn').disabled = true;
            document.getElementById('pauseBtn').disabled = false;
            document.getElementById('stopBtn').disabled = false;

            // A velocidade base de 1000ms é dividida pelo multiplicador
            const intervalTime = 1000 / playbackMultiplier;

            playbackInterval = setInterval(() => {
                if (currentWaypointIndex < waypoints.length - 1) {
                    currentWaypointIndex++;
                    document.getElementById('timelineSlider').value = currentWaypointIndex;
                    timelineUpdate(currentWaypointIndex);
                } else {
                    playbackStop();
                }
            }, intervalTime);
        }

        function playbackPause() {
            isPlaying = false;
            clearInterval(playbackInterval);

            document.getElementById('playBtn').disabled = false;
            document.getElementById('pauseBtn').disabled = true;
        }

        function playbackStop() {
            isPlaying = false;
            clearInterval(playbackInterval);

            currentWaypointIndex = 0;
            document.getElementById('timelineSlider').value = 0;
            timelineUpdate(0);

            document.getElementById('playBtn').disabled = false;
            document.getElementById('pauseBtn').disabled = true;
            document.getElementById('stopBtn').disabled = true;
        }

        function updateMapStatus(message) {
            document.getElementById('mapStatus').textContent = message;
        }

        function changeOrganization(select) {
            showToast('Organização alterada: ' + select.value);
        }

        function showToast(msg, duration = 3000) {
            const toast = document.getElementById('toast');
            toast.innerHTML = msg;
            toast.style.display = 'block';

            clearTimeout(window._toastTimer);
            window._toastTimer = setTimeout(() => {
                toast.style.display = 'none';
            }, duration);
        }
