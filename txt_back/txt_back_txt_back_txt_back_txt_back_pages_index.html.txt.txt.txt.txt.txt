        // Configurar tema inicial baseado no localStorage
        document.addEventListener('DOMContentLoaded', () => {
            const savedTheme = localStorage.getItem('theme') || 'dark';
            document.body.className = `theme-${savedTheme}`;
        });

        // Ouvir mudanças de tema da navegação
        document.addEventListener('themeChanged', (e) => {
            document.body.className = `theme-${e.detail.theme}`;

            // Atualizar cor do "7" na logo quando o tema mudar
            const sevenSpan = document.querySelector('.brand-text span');
            if (sevenSpan) {
                sevenSpan.style.color = e.detail.theme === 'dark' ? '#ffffff' : '#000000';
            }
        });
